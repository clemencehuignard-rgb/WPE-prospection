Créer la table (SQL Editor) :
Colle et exécute le SQL ci-dessous dans SQL Editor de Supabase :
create table prospects (
  id bigserial primary key,
  nom text not null,
  segment text default 'luxury_premium',
  pays text default 'France',
  contact text,
  email text,
  tel text,
  linkedin text,
  score integer default 0,
  tier text default '2',
  statut text default 'a-contacter',
  notes text,
  site text,
  next_contact date,
  created_at timestamptz default now()
);
alter table prospects enable row level security;
create policy "public_all" on prospects for all using (true) with check (true);
